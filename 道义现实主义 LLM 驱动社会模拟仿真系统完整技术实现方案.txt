│                                                                                                                      │
│ core/llm_engine.py                                                                                                   │
│ """                                                                                                                  │
│ LLM决策引擎 - 封装SiliconFlow API调用                                                                                │
│ """                                                                                                                  │
│ from typing import Any, Dict, List, Optional                                                                         │
│ from dataclasses import dataclass                                                                                    │
│ import os                                                                                                            │
│ from openai import OpenAI                                                                                            │
│ from loguru import logger                                                                                            │
│                                                                                                                      │
│                                                                                                                      │
│ @dataclass                                                                                                           │
│ class LLMConfig:                                                                                                     │
│     """LLM配置"""                                                                                                    │
│     base_url: str                                                                                                    │
│     api_key: str                                                                                                     │
│     model: str                                                                                                       │
│     temperature: float = 0.7                                                                                         │
│     max_tokens: int = 2000                                                                                           │
│     timeout: int = 60                                                                                                │
│                                                                                                                      │
│                                                                                                                      │
│ class LLMEngine:                                                                                                     │
│     """LLM决策引擎"""                                                                                                │
│                                                                                                                      │
│     def __init__(self, config: Optional[LLMConfig] = None):                                                          │
│         if config is None:                                                                                           │
│             config = LLMConfig(                                                                                      │
│                 base_url=os.getenv("SILICONFLOW_BASE_URL", "https://api.siliconflow.cn/v1"),                         │
│                 api_key=os.getenv("SILICONFLOW_API_KEY", ""),                                                        │
│                 model=os.getenv("SILICONFLOW_MODEL", "deepseek-ai/DeepSeek-V2.5")                                    │
│             )                                                                                                        │
│                                                                                                                      │
│         self.config = config                                                                                         │
│         self.client = OpenAI(                                                                                        │
│             base_url=config.base_url,                                                                                │
│             api_key=config.api_key                                                                                   │
│         )                                                                                                            │
│         logger.info(f"LLM Engine initialized with model: {config.model}")                                            │
│                                                                                                                      │
│     def chat_completion(                                                                                             │
│         self,                                                                                                        │
│         messages: List[Dict[str, str]],                                                                              │
│         temperature: Optional[float] = None,                                                                         │
│         max_tokens: Optional[int] = None,                                                                            │
│         stream: bool = False                                                                                         │
│     ) -> Any:                                                                                                        │
│         """                                                                                                          │
│         发送聊天补全请求                                                                                             │
│                                                                                                                      │
│         Args:                                                                                                        │
│             messages: 消息列表                                                                                       │
│             temperature: 温度参数                                                                                    │
│             max_tokens: 最大token数                                                                                  │
│             stream: 是否流式输出                                                                                     │
│                                                                                                                      │
│         Returns:                                                                                                     │
│             API响应                                                                                                  │
│         """                                                                                                          │
│         try:                                                                                                         │
│             response = self.client.chat.completions.create(                                                          │
│                 model=self.config.model,                                                                             │
│                 messages=messages,                                                                                   │
│                 temperature=temperature or self.config.temperature,                                                  │
│                 max_tokens=max_tokens or self.config.max_tokens,                                                     │
│                 stream=stream                                                                                        │
│             )                                                                                                        │
│                                                                                                                      │
│             if stream:                                                                                               │
│                 return response                                                                                      │
│             else:                                                                                                    │
│                 return response.choices[0].message.content                                                           │
│                                                                                                                      │
│         except Exception as e:                                                                                       │
│             logger.error(f"LLM API call failed: {e}")                                                                │
│             raise                                                                                                    │
│                                                                                                                      │
│     def function_call(                                                                                               │
│         self,                                                                                                        │
│         messages: List[Dict[str, str]],                                                                              │
│         functions: List[Dict[str, Any]],                                                                             │
│         function_call: str = "auto"                                                                                  │
│     ) -> Dict[str, Any]:                                                                                             │
│         """                                                                                                          │
│         函数调用 - 用于约束智能体行为选择                                                                            │
│                                                                                                                      │
│         Args:                                                                                                        │
│             messages: 消息列表                                                                                       │
│             functions: 可用函数列表                                                                                  │
│             function_call: 函数调用模式                                                                              │
│                                                                                                                      │
│         Returns:                                                                                                     │
│             函数调用结果                                                                                             │
│         """                                                                                                          │
│         try:                                                                                                         │
│             response = self.client.chat.completions.create(                                                          │
│                 model=self.config.model,                                                                             │
│                 messages=messages,                                                                                   │
│                 functions=functions,                                                                                 │
│                 function_call=function_call,                                                                         │
│                 temperature=0.3  # 较低温度确保一致性行为                                                            │
│             )                                                                                                        │
│                                                                                                                      │
│             return response.choices[0].message                                                                       │
│                                                                                                                      │
│         except Exception as e:                                                                                       │
│             logger.error(f"Function call failed: {e}")                                                               │
│             raise                                                                                                    │
│                                                                                                                      │
│     def stream_chat_completion(                                                                                      │
│         self,                                                                                                        │
│         messages: List[Dict[str, str]],                                                                              │
│         callback: callable                                                                                           │
│     ) -> None:                                                                                                       │
│         """                                                                                                          │
│         流式聊天补全 - 用于实时响应                                                                                  │
│                                                                                                                      │
│         Args:                                                                                                        │
│             messages: 消息列表                                                                                       │
│             callback: 回调函数处理流式输出                                                                           │
│         """                                                                                                          │
│         response = self.client.chat.completions.create(                                                              │
│             model=self.config.model,                                                                                 │
│             messages=messages,                                                                                       │
│             temperature=self.config.temperature,                                                                     │
│             max_tokens=self.config.max_tokens,                                                                       │
│             stream=True                                                                                              │
│         )                                                                                                            │
│                                                                                                                      │
│         for chunk in response:                                                                                       │
│             if chunk.choices and chunk.choices[0].delta.content:                                                     │
│                 callback(chunk.choices[0].delta.content)                                                             │
│             if chunk.choices and chunk.choices[0].delta.reasoning_content:                                           │
│                 callback(f"[Reasoning]: {chunk.choices[0].delta.reasoning_content}")                                 │
│                                                                                                                      │
│ 1.3 核心数据模型 - 领导类型                                                                                          │
│                                                                                                                      │
│ models/leadership_type.py                                                                                            │
│ """                                                                                                                  │
│ 政治领导类型模型 - 核心自变量                                                                                        │
│ 基于阎学通论文的政府道义分类标准                                                                                     │
│ """                                                                                                                  │
│ from enum import Enum                                                                                                │
│ from dataclasses import dataclass                                                                                    │
│ from typing import List, Set                                                                                         │
│                                                                                                                      │
│                                                                                                                      │
│ class LeadershipType(Enum):                                                                                          │
│     """政治领导类型"""                                                                                               │
│     WANGDAO = "wangdao"      # 王道型                                                                                │
│     HEGEMON = "hegemon"       # 霸权型                                                                               │
│     QIANGQUAN = "qiangquan"   # 强权型                                                                               │
│     HUNYONG = "hunyong"       # 昏庸型                                                                               │
│                                                                                                                      │
│                                                                                                                      │
│ @dataclass                                                                                                           │
│ class LeadershipProfile:                                                                                             │
│     """领导类型画像 - 刚性定义各类领导的核心属性"""                                                                  │
│                                                                                                                      │
│     leadership_type: LeadershipType                                                                                  │
│                                                                                                                      │
│     # 道义标准                                                                                                       │
│     moral_standard: str                                                                                              │
│                                                                                                                      │
│     # 利益偏好排序（从高到低）                                                                                       │

│     def add_history(self, record: Dict[str, Any]) -> None:                                                           │
│         """添加历史记录"""                                                                                           │

│                                                                                                                      │

│             "type": "response",                                                                                      │
│             "action": action,                                                                                        │
│             "response": parsed_response                                                                              │

│                                                                                                                      │

│ from core.llm_engine import LLMEngine                                                                                │

│             ]                                                                                                        │

│         return risks                                                                                                 │

│                                                                                                                      │

│                 "action": "align",                                                                                   │
│                 "target": best_option["country"],                                                                    │
│                 "reasoning": f"跟随{best_option['country']}可获得{best_option['benefits']}"                          │
│             }                                                                                                        │

│             return {                                                                                                 │
│                 "action": "remain_neutral",                                                                          │
│                 "reasoning": "所有大国选项风险过高，保持中立"                                                        │
│             }                                                                                                        │
│                                                                                                                      │

│         # 随机危机事件                                                                                               │

│                                                                                                                      │

│                                                                                                                      │

│                                                                                                                      │

│         计算道义水平总指数                                                                                           │
│                                                                                                                      │
│         综合下属6个二级指标加权计算                                                                                  │
│         """                                                                                                          │
│         # 战略信誉度 (25%)                                                                                           │
│         credibility = moral_scores["strategic_credibility"]                                                          │
│                                                                                                                      │
│         # 规范遵守度 (25%)                                                                                           │
│         compliance = moral_scores["norm_compliance"]                                                                 │
│                                                                                                                      │
│         # 双重标准指数 (反向，20%)                                                                                   │
│         double_standard = 100 - moral_scores["double_standard"]                                                      │
│                                                                                                                      │
│         # 暴力偏好度 (反向，15%)                                                                                     │
│         violence = 100 - moral_scores["violence_preference"]                                                         │
│                                                                                                                      │
│         # 公共利益贡献度 (15%)                                                                                       │
│         public_interest = moral_scores["public_interest_contribution"]                                               │
│                                                                                                                      │
│         total = (                                                                                                    │
│             credibility * 0.25 +                                                                                     │
│             compliance * 0.25 +                                                                                      │
│             double_standard * 0.20 +                                                                                 │
│             violence * 0.15 +                                                                                        │
│             public_interest * 0.15                                                                                   │
│         )                                                                                                            │
│                                                                                                                      │
│         return total                                                                                                 │

│     def check_order_evolution(                                                                                       │
│         self,                                                                                                        │
│         old_state: Dict[str, Any],                                                                                   │
│         new_state: Dict[str, Any]                                                                                    │
│     ) -> Tuple[bool, str]:                                                                                           │
│         """                                                                                                          │
│         检查国际秩序演变                                                                                             │
│                                                                                                                      │
│         规则（刚性）：                                                                                               │
│         1. 仅改变权力分配：仅引发国际格局变化，不改变国际秩序类型                                                    │
│         2. 仅改变价值观与国际规范：改变国际秩序的规范特征与类型                                                      │
│         3. 同时改变价值观、规范、权力分配：国际秩序必然发生类型变化                                                  │
│                                                                                                                      │
│         Args:                                                                                                        │
│             old_state: 旧状态                                                                                        │
│             new_state: 新状态                                                                                        │
│                                                                                                                      │
│         Returns:                                                                                                     │
│             (是否改变秩序类型, 原因说明)                                                                             │
│         """                                                                                                          │
│         power_changed = self._power_distribution_changed(old_state, new_state)                                       │
│         values_changed = self._values_changed(old_state, new_state)                                                  │
│         norms_changed = self._norms_changed(old_state, new_state)                                                    │
│                                                                                                                      │
│         # 情况1：仅权力改变                                                                                          │
│         if power_changed and not values_changed and not norms_changed:                                               │
│             return False, "仅权力分配改变，仅影响国际格局"                                                           │
│                                                                                                                      │
│         # 情况2：仅价值观或规范改变                                                                                  │
│         if not power_changed and (values_changed or norms_changed):                                                  │
│             return True, "价值观或国际规范改变，引发秩序类型变化"                                                    │
│                                                                                                                      │
│         # 情况3：三者同时改变                                                                                        │
│         if power_changed and values_changed and norms_changed:                                                       │
│             return True, "权力、价值观、规范同时改变，必然引发秩序变化"                                              │
│                                                                                                                      │
│         # 情况4：权力+规范/价值观改变                                                                                │
│         if power_changed and (values_changed or norms_changed):                                                      │
│             return True, "权力与规范/价值观共同改变，引发秩序变化"                                                   │
│                                                                                                                      │
│         return False, "无显著变化"                                                                                   │
│                                                                                                                      │
│     def _power_distribution_changed(                                                                                 │
│         self,                                                                                                        │
│         old_state: Dict[str, Any],                                                                                   │
│         new_state: Dict[str, Any]                                                                                    │
│     ) -> bool:                                                                                                       │
│         """检查权力分配是否改变"""                                                                                   │
│         old_power = old_state.get("power_distribution", {})                                                          │
│         new_power = new_state.get("power_distribution", {})                                                          │
│                                                                                                                      │
│         if old_power != new_power:                                                                                   │
│             return True                                                                                              │
│                                                                                                                      │
│         # 检查国家实力变化                                                                                           │
│         old_agents = old_state.get("agents", {})                                                                     │
│         new_agents = new_state.get("agents", {})                                                                     │
│                                                                                                                      │
│         for name in set(old_agents.keys()) | set(new_agents.keys()):                                                 │
│             old_capability = old_agents.get(name, {}).get("capability", {})                                          │
│             new_capability = new_agents.get(name, {}).get("capability", {})                                          │
│                                                                                                                      │
│             if old_capability != new_capability:                                                                     │
│                 return True                                                                                          │
│                                                                                                                      │
│         return False                                                                                                 │
│                                                                                                                      │
│     def _values_changed(                                                                                             │

│         new_state: Dict[str, Any]                                                                                    │

│         old_values = old_state.get("dominant_values", set())                                                         │
│         new_values = new_state.get("dominant_values", set())                                                         │
│                                                                                                                      │
│         return old_values != new_values                                                                              │
│                                                                                                                      │
│     def _norms_changed(                                                                                              │

│         new_norms = new_state.get("norms", [])                                                                       │
│                                                                                                                      │
│         if len(old_norms) != len(new_norms):                                                                         │
│             return True                                                                                              │
│                                                                                                                      │
│         # 比较规范的权威度                                                                                           │
│         for i, old_norm in enumerate(old_norms):                                                                     │
│             if i < len(new_norms):                                                                                   │
│                 if old_norm.get("authority") != new_norms[i].get("authority"):                                       │
│                     return True                                                                                      │
│                                                                                                                      │
│         return False                                                                                                 │
│                                                                                                                      │
│     def determine_order_type(                                                                                        │
│         self,                                                                                                        │
│         agents: Dict[str, Any],                                                                                      │
│         moral_scores: Dict[str, Dict[str, float]],                                                                   │
│         global_conflict_level: float                                                                                 │

│         """                                                                                                          │
│         判定国际秩序类型                                                                                             │
│                                                                                                                      │
│         规则：                                                                                                       │
│         1. 王道型秩序：高道义水平、低冲突、高公共产品供给                                                            │
│         2. 霸权型秩序：中等道义、选择性规范、霸权主导                                                                │
│         3. 强权型秩序：低道义、高冲突、强制手段                                                                      │
│         4. 失序型秩序：无有效领导、极高冲突、组织瘫痪                                                                │
│         """                                                                                                          │
│         # 计算全球平均道义水平                                                                                       │
│         if moral_scores:                                                                                             │
│             avg_moral = sum(                                                                                         │
│                 scores.get("total_moral_level", 50)                                                                  │
│                 for scores in moral_scores.values()                                                                  │
│             ) / len(moral_scores)                                                                                    │
│         else:                                                                                                        │
│             avg_moral = 50.0                                                                                         │
│                                                                                                                      │
│         # 判定秩序类型                                                                                               │
│         if global_conflict_level > 80:                                                                               │
│             return "disordered"  # 失序型秩序                                                                        │
│                                                                                                                      │
│         if avg_moral >= 70:                                                                                          │
│             return "wangdao_order"  # 王道型秩序                                                                     │
│         elif avg_moral >= 45:                                                                                        │
│             return "hegemon_order"  # 霸权型秩序                                                                     │
│         elif avg_moral >= 25:                                                                                        │
│             return "qiangquan_order"  # 强权型秩序                                                                   │
│         else:                                                                                                        │
│             return "disordered"  # 失序型秩序                                                                        │
